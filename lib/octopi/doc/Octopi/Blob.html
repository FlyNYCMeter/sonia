def self.find(user, repo, sha, path = nil)
  user = user.to_s
  repo = repo.to_s
  self.validate_args(sha => :sha, user => :user)
  if path
    super [user, repo, sha, path]
  else
    Api.api.get_raw(path_for(:resource), {:id => [user, repo, sha].join('/')})
  end  
end
