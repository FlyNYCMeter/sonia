def self.all(opts={})
  user, repo = gather_details(opts)
  self.validate_args(user => :user, repo => :repo)
  BranchSet.new(find_plural([user, repo, 'branches'], :resource)) do |i|
    { :name => i.first, :hash => i.last }
  end
end
